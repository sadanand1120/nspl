def is_dropoff(pixel_loc):
    if terrain(pixel_loc) == 3:
        if frontal_distance_entrance(pixel_loc) > 0.0:
            return True
    elif terrain(pixel_loc) == 0:
        if frontal_distance_entrance(pixel_loc) > 0.0 and distance_to_person(pixel_loc) > 2.1 and in_the_way(pixel_loc) == 0 and distance_to_bush(pixel_loc) > 0.2 and frontal_distance_staircase(pixel_loc) > 0.0 and distance_to_car(pixel_loc) > 3.0:
            return True
    return False
