def is_dropoff(pixel_loc):
    if terrain(pixel_loc) == 3:
        if frontal_distance_entrance(pixel_loc) > 0.0 and distance_to_wall(pixel_loc) > 0.40165:
            return True
    elif terrain(pixel_loc) == 0:
        if distance_to_person(pixel_loc) > 2.1 and distance_to_pole(pixel_loc) > 0.0 and in_the_way(pixel_loc) == 0 and frontal_distance_entrance(pixel_loc) > 0.0 and distance_to_bush(pixel_loc) > 0.2 and distance_to_barricade(pixel_loc) > 0.0 and distance_to_tree(pixel_loc) > 0.0 and distance_to_board(pixel_loc) > 2.5 and frontal_distance_staircase(pixel_loc) > 0.0 and distance_to_car(pixel_loc) > 3.0:
            return True
    return False
